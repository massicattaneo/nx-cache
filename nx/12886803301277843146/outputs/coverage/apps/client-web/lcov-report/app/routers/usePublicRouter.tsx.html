import { useCallback, useRef } from "react"
import { Outlet, RouteObject } from "react-router-dom"
import { AuthenticateWithRedirectCallback } from "@clerk/clerk-react"
import { environment } from "@rp/common/environment"
import { DEEP_LINK_TOOL_HTTPS_URL, MOBILE_ONBOARDING_BANNER_KEY } from "@rp/common/utils"
import { MobileBanner } from "@rp/react/web/common/composite-components"
import { useDeviceInfo, useMobileNativePopup } from "@rp/react/web/common/hooks"
import { useCustomerIOPageViewEvents } from "@rp/react/web/common/providers"
import * as URLS from "@rp/react/web/router"
import { RouterErrorPage } from "../pages/ErrorPage/RouterErrorPage"
import { LoginPage } from "../pages/Login"
import { PasswordRecoveryPage } from "../pages/PasswordRecovery"
import { SSOLoginPage } from "../pages/SSOLogin"
import { SignUpPage } from "../pages/SignUp"
import { Invitation } from "../pages/invitation"
import { createBrowserRouterWithBasename } from "./createBrowserRouterWithBasename"
import { CommonWrapper } from "./wrappers/CommonWrapper"
 
function UnauthenticatedWrapper() {
  useCustomerIOPageViewEvents()
  const { isAndroid, isIOS, isSafari } = useDeviceInfo()
  const { displayState, saveLSAndHideComponent } = useMobileNativePopup(
    MOBILE_ONBOARDING_BANNER_KEY,
  )
  const isMobileModalVisible = (isIOS || isAndroid) && !isSafari && displayState
 
  return (
    <CommonWrapper>
      <MobileBanner
        deepLinkURL={DEEP_LINK_TOOL_HTTPS_URL}
        isVisible={isMobileModalVisible}
        onClose={saveLSAndHideComponent}
      />
      <Outlet />
    </CommonWrapper>
  )
}
 
function usePublicRouter() {
  const redirectRef = useRef(window.location.href)
 
  // Handle errors and report them to sentry from the error boundary
  const onError = useCallback((error: unknown) => reportError(error), [])
 
  const routes: Array<RouteObject> = [
    { path: URLS.LOGIN_ROUTE, element: <LoginPage redirect={redirectRef.current} /> },
    { path: URLS.SSO_LOGIN_ROUTE, element: <SSOLoginPage redirect={redirectRef.current} /> },
    { path: URLS.RESET_PASSWORD_ROUTE, element: <PasswordRecoveryPage /> },
    { path: URLS.SIGN_UP_ROUTE, element: <SignUpPage /> },
    { path: URLS.INVITATION_ROUTE, element: <Invitation /> },
    {
      path: "/sso-callback",
      element: (
        <AuthenticateWithRedirectCallback
          afterSignInUrl={environment.app.url}
          afterSignUpUrl={environment.app.url}
        />
      ),
    },
    { path: "*", element: <LoginPage redirect={redirectRef.current} /> },
  ]
 
  return createBrowserRouterWithBasename([
    {
      element: <UnauthenticatedWrapper />,
      errorElement: <RouterErrorPage onError={onError} />,
      children: routes,
    },
  ])
}
 
export { usePublicRouter }
